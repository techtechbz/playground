## c++

```c++
#include <iostream>
#include <cmath>
using namespace std;

int main() {
  // c++では**演算子は使用できない
//   cout << 3 ** 2 << endl;
  cout << pow(3, 2) << endl;
  cout << pow(3, 2) << endl;
  cout << pow(1.5, 2) << endl;
  cout << pow(4, 1.5) << endl;
  cout << pow(0.4, 1.5) << endl;
  cout << pow(-3, 2) << endl;
  cout << pow(3, -2) << endl;
  cout << pow(-3, -2) << endl;
  cout << pow(3, 0) << endl;
  cout << pow(0, 2) << endl;
}
```

## Python

```python
print(3 ** 2)
print(pow(3, 2))
print(3 ** 2)
print(1.5 ** 2)
print(4 ** 1.5)
print(0.4 ** 1.5)
# 負の数の累乗を計算したい場合は、負の数をパーレンで囲む。
# そうでなければ、正の数として計算した値にマイナスが付与される。
print((-3) ** 2)
print(3 ** -2)
print((-3) ** -2)
print(3 ** 0)
print(0 ** 2)

```
